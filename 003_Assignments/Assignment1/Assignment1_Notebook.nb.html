title_df <- read.csv("titles.csv", as.is = c("genres"))

split_gernes <- function(x){
  x <- gsub("'", "", x)
  return(str_split(substr(x, 2, nchar(x)-1), ", "))
}
title_df$genres2 <- sapply(title_df$genres, FUN=split_gernes)
genres_categories <- Reduce(union, title_df$genres2)

genres_count <- function(genres.list){
  
  genres_vec <- function(x){
  if(x %in% genres.list){
    return(1)
  }
  else {
    return(0)
  }
  }
  return(genres_vec)
}
